Create tables

    CREATE TABLE Actor (
        Act_id int PRIMARY KEY,
        Act_FName varchar(50),
        Act_LName varchar(50),
        Act_Gender varchar(10),
        Mov_id int,
        Salary numeric(10, 2)
    );
    
    CREATE TABLE Movies (
        Mov_id int PRIMARY KEY,
        Mov_Title varchar(100),
        Mov_year int,
        Mov_Lang varchar(50)
    );

Create a view with actor details

    CREATE VIEW Actor_View AS
    SELECT Act_id, CONCAT(Act_FName, ' ', Act_LName) AS Full_Name, Act_Gender FROM Actor;
    
     
Create a function to calculate the incremented salary

    DELIMITER //
    CREATE FUNCTION CalculateIncrementedSalary(actor_id INT, increment_amount NUMERIC) RETURNS NUMERIC
    BEGIN
        DECLARE current_salary NUMERIC;
        DECLARE new_salary NUMERIC;
        
        SELECT Salary INTO current_salary FROM Actor WHERE Act_id = actor_id;
        SET new_salary = current_salary + increment_amount;
        
        RETURN new_salary;
    END //
    DELIMITER ;
    
     
Display the incremented salary

    SELECT Act_id, CONCAT(Act_FName, ' ', Act_LName) AS Full_Name, CalculateIncrementedSalary(Act_id, 5000) AS Incremented_Salary FROM Actor;

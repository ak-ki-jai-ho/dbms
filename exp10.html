Create a trigger to store the details of Actors with current date before delete operation on actor table

CREATE TABLE Deleted_Actors (
    Act_id int,
    Act_FName varchar(50),
    Act_LName varchar(50),
    Act_Gender varchar(10),
    Date_deleted date
);

DELIMITER //

CREATE TRIGGER before_actor_delete
BEFORE DELETE ON Actor
FOR EACH ROW
BEGIN
    INSERT INTO Deleted_Actors (Act_id, Act_FName, Act_LName, Act_Gender, Date_deleted)
    VALUES (OLD.Act_id, OLD.Act_FName, OLD.Act_LName, OLD.Act_Gender, CURDATE());
END //

DELIMITER ;

 
Perform delete operation on Actor table and display the audit table

DELETE FROM Actor;

SELECT * FROM Deleted_Actors;

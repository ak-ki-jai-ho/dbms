Create tables

                CREATE TABLE Employee (
                    SSN varchar(11) PRIMARY KEY,
                    Name varchar(100),
                    Address varchar(200),
                    Sex varchar(10),
                    Salary numeric(10, 2),
                    SuperSSN varchar(11),
                    DNo int
                );
                
                CREATE TABLE Department (
                    DNo int PRIMARY KEY,
                    DName varchar(100),
                    MgrSSN varchar(11),
                    MgrStartDate date
                );

                SELECT * FROM Employee
                ORDER BY Salary DESC;

                UPDATE Employee

Create table with integrity constraints

                CREATE TABLE Book (
                    ISBN varchar(40) NULL,
                    Book_Title varchar(30) NOT NULL,
                    Book_Code varchar(30) PRIMARY KEY,
                    Book_Price numeric(8, 2) NOT NULL CHECK (Book_Price >= 0),
                    Author_ID char(1) NOT NULL,
                    Paperback char(1) DEFAULT 'Y' CHECK (Paperback IN ('Y', 'N'))
                );

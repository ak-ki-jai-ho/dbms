Create tables

CREATE TABLE Book (
    ISBN varchar(40),
    Bk_Title varchar(100),
    Author_ID int,
    Bk_Type varchar(50),
    Price numeric(8, 2)
);

CREATE TABLE Author (
    Author_ID int PRIMARY KEY,
    Author_fname varchar(50),
    Author_lname varchar(50),
    City varchar(50),
    Contact_No varchar(15)
);

Update the city of the author whose last name is “Patil”

SET City = 'Pune'
WHERE Author_lname = 'Patil';

 
Rename the Book table to Book_details and describe the table

ALTER TABLE Book RENAME TO Book_details;
DESCRIBE Book_details;

 
Find all the books written by author whose Id is 129

SELECT * FROM Book_details
WHERE Author_ID = 129;

 
Find the title of the most expensive book of type “Science” written by author “Elmasri”

SELECT Bk_Title
FROM Book_details
WHERE Bk_Type = 'Science' AND Author_ID = (
    SELECT Author_ID FROM Author WHERE Author_lname = 'Elmasri'
)
ORDER BY Price DESC
LIMIT 1;

 
Find the number of authors from ‘Thane’ city

SELECT COUNT(*) AS Num_Authors_Thane FROM Author WHERE City = 'Thane';

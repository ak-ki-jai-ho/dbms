Create tables

CREATE TABLE Employee (
    SSN varchar(11) PRIMARY KEY,
    Name varchar(100),
    Address varchar(200),
    Sex varchar(10),
    Salary numeric(10, 2),
    SuperSSN varchar(11),
    DNo int
);

CREATE TABLE Department (
    DNo int PRIMARY KEY,
    DName varchar(100),
    MgrSSN varchar(11),
    MgrStartDate date
);

Find the employee who earns the lowest salary of all the departments

SELECT * FROM Employee
WHERE Salary = (SELECT MIN(Salary) FROM Employee);

Create tables

CREATE TABLE Actor (
    Act_id int PRIMARY KEY,
    Act_FName varchar(50),
    Act_LName varchar(50),
    Act_Gender varchar(10),
    Mov_id int
);

CREATE TABLE Movies (
    Mov_id int PRIMARY KEY,
    Mov_Title varchar(100),
    Mov_year int,
    Mov_Lang varchar(50)
);

Arrange the movies in descending order of year in which released

SELECT * FROM Movies
ORDER BY Mov_year DESC;

 
List all the movies whose language is not ‘Hindi’

SELECT * FROM Movies
WHERE Mov_Lang != 'Hindi';

 
Find the length of the Movie titles

SELECT Mov_Title, LENGTH(Mov_Title) AS Title_Length FROM Movies;

 
List all the actor’s name which starts with character ‘M’

SELECT Act_FName, Act_LName FROM Actor
WHERE Act_FName LIKE 'M%';

 
Merge the actors’ first name and last name and display it

SELECT CONCAT(Act_FName, ' ', Act_LName) AS Full_Name FROM Actor;

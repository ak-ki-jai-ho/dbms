Create tables

CREATE TABLE Director (
    Dir_id int PRIMARY KEY,
    Dir_FName varchar(50),
    Dir_LName varchar(50),
    Dir_city varchar(100),
    Dir_Phone varchar(15)
);

CREATE TABLE Movies (
    Mov_id int PRIMARY KEY,
    Mov_Title varchar(100),
    Mov_year int,
    Mov_Lang varchar(50),
    Dir_id int,
    FOREIGN KEY (Dir_id) REFERENCES Director(Dir_id)
);

Create a trigger to store the details of Directors with current date before delete operation on director table

CREATE TABLE Deleted_Directors (
    Dir_id int,
    Dir_FName varchar(50),
    Dir_LName varchar(50),
    Dir_Phone varchar(15),
    Date_deleted date
);

DELIMITER //
CREATE TRIGGER before_director_delete
BEFORE DELETE ON Director
FOR EACH ROW
BEGIN
    INSERT INTO Deleted_Directors (Dir_id, Dir_FName, Dir_LName, Dir_Phone, Date_deleted)
    VALUES (OLD.Dir_id, OLD.Dir_FName, OLD.Dir_LName, OLD.Dir_Phone, CURDATE());
END //
DELIMITER ;

    prod_cost numeric(10, 2),

Create a function to calculate the profit of products

CREATE FUNCTION CalculateProfit(cost numeric, price numeric) RETURNS numeric
BEGIN
    DECLARE profit numeric;
    SET profit = price - cost;
    RETURN profit;
END;

 
Display the profit calculated

SELECT prod_name, CalculateProfit(prod_cost, prod_price) AS Profit FROM Products;
